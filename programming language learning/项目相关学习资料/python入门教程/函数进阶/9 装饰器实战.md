设计一个系统可以登录之后进行增删改查的操作系统
```python
def login_verify(fn):  
    def inner(*args, **kwargs):  
        # 这里完成登录校验  
        print("not login now")  
        while 1:  
            username = input(">>>")  
            password = input(">>>")  
            if username == "admin" and password == "123":  
                print("login successfully")  
                break  
            else:  
                print("login false")  
  
        ret = fn(*args, **kwargs)  
        # 后续程序执行  
        return ret  
    return inner  
  
@login_verify  
def add():  
    print("add workers information")  
  
@login_verify  
def delete():  
    print("delete workers info")  
  
@login_verify  
def upd():  
    print("update workers info")  
  
@login_verify  
def search():  
    print("search workers info")  
  
add()  
upd()  
search()  
delete()
```
这样操作的结果是每次增删改查都需要重新登录，这样很麻烦
```python
login_flag = False  
  
def login_verify(fn):  
    def inner(*args, **kwargs):  
        global login_flag  
        if login_flag == False:  
  
            # 这里完成登录校验  
            print("not login now")  
            while 1:  
                username = input(">>>")  
                password = input(">>>")  
                if username == "admin" and password == "123":  
                    print("login successfully")  
                    login_flag = True  
                    break                
                else:  
                    print("login false")  
  
        ret = fn(*args, **kwargs)  
        # 后续程序执行  
        return ret  
    return inner  
  
@login_verify  
def add():  
    print("add workers information")  
  
@login_verify  
def delete():  
    print("delete workers info")  
  
@login_verify  
def upd():  
    print("update workers info")  
  
@login_verify  
def search():  
    print("search workers info")  
  
add()  
upd()  
search()  
delete()
```
这样操作的话就可以记录当下的状态，就不需要增删改查每次都操作
