```python
# 1. 注意读和写的区别  
  
class Person(object):  
    country = "中国"  
  
    def __init__(self,name,age):  
        self.name = name  
        self.age = age  
  
    def show(self):  
        message = f"{self.country}-{self.name}-{self.age}"  
        print(message)  
  
print(Person.country) # 中国  
  
p1 = Person("wupeiqi", 20)  
print(p1.name) # wupeiqi  
print(p1.age) # 20  
print(p1.country) # 中国  
p1.show() # 中国-wupeiqi-20  
  
p1.name = "root" # 在对象p1中将name重置为root  
p1.num = 19 # 在对象p1中新增一个实例变量num=19  
p1.country = 'China' # 在p1中新增一个实例变量country="China"  
# 注意,这里跟类变量没有关系, 因为country不是定义在init中的,所以p1中其实没有country  
# 这时候再print country,就不会去类当中找country变量了  
# 给谁设置就是在什么内存中改值  
  
Person.country = "美国"  
# 这时候,实例化的对象找country,如果没有定义,就会找到Person中改掉的变量,也就是美国  
# 在设置的时候不会跨关系进行修改
```

```python
  
class Base(object):  
    country = "中国"  
  
class Person(Base):  
  
    def __init__(self,name,age):        self.name = name        self.age = age  
    def show(self):        message = f"{Person.country}-{self.name}-{self.age}"        # message = f"{self.country}-{self.name}-{self.age}"        print(message)  
# 读  
print(Base.country) # 中国  
print(Person.country) # 自己里面没有找base,是中国  
  
obj = Person('wupeiqi',19)  
print(obj.country) # 中国  
  
# 写  
Base.country = "China"  
# 这里Base 的country就变了,后边所有的因为都是在Base找,所以也会变  
Person.country = "泰国"  
# 这里是在Person中添加一个新的变量 country = "泰国"  
obj.country = "日本"  
# 在利用person类创建的obj对象中创建了一个"日本",只在这个obj空间中有这个对应关系
```

```python
class Parent(object):  
    x = 1  
  
class Child1(Parent):  
    pass  
  
class Child2(Parent):  
    pass  
  
print(Parent.x, Child1.x, Child2.x) # 1,1,1  
  
Child1.x = 2  
# 这里添加了一个类变量 x = 2在Child1中  
print(Parent.x, Child1.x, Child2.x) # 1,2,1  
  
Parent.x = 3  
# 这里因为Child1已经有x了,改了Parent对Child1没有关系,但是Child2没有改,所以还是跟Parent一起  
print(Parent.x, Child1.x, Child2.x) # 3,2,3
```
