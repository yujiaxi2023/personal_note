```python
"""
结合下面两个变量header 和 stock_dict实现注意输出股票信息,格式如下
	SH601778,股票名称:中国荆轲,当前价:6.29,涨跌额:+1.92
	...
"""
header = ['股票名称', '当前价', '涨跌额']

stock_dict = {
	'SH601778':['中国荆轲', '6.29', '+1.92'],
	'SH688566':['吉贝尔', '52.66', '+6.96'],
	'SH688268':['华特气体', '88.80', '+11.72'],
	'SH600734':['实达集团', '2.60', '+0.24']
}
```

```python
for key, value in stock_dict.items():
	# print(key, value)

	text_list = []
	for index in range(len(value)):
		text = f"{header[index]}:{value[index]}"
		text_list.append(text)
	
	data = ",".join(text_list)

	result = f"{key},{data}."
```
