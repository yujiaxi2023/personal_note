```python
# python中支持一个特殊的语法：@函数名  
"""  
python中的特殊语法，某个函数的上方使用：  
  
@函数名：  
def xxx():  
    pass  
python内部会自动执行这个函数名  
并且将下面的函数名作为参数传入执行的这个函数  
并且执行完后将结果赋值给xxx  
xxx = 函数名(xxx)  
"""  
def outer(origin):  
    def inner():  
        print("before")  
        res = origin() # 这样就会执行之前存储的一个func函数  
        print("after")  
        return res  
  
    return inner  
  
@outer  
def func():  
    print("我是func函数")  
    value = (11, 22, 33, 44)  
    return value  
  
  
func = outer(func)  # func就是原来的函数传进去了，就在outer中维护了一个原来的func函数  
result = func()  # operating inner  
print(result)
"""
before
before
我是func函数
after
after
(11, 22, 33, 44)
"""
```
这个结果不对
