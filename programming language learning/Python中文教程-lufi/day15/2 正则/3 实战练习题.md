```python
import re  
import requests  
  
from bs4 import BeautifulSoup  
  
# 1. 利用正则匹配qq号码  
text = "806986756"  
data_list = re.findall("[1-9]\d{4,}",text)  
"""  
不可以用0开头  
需要至少是4位数  
"""  
print(data_list) # ['806986756']  
  
  
# 2. 身份证号码  
text = "dsf130429191912015219k1304291959121952Xkk"  
data_list = re.findall("\d{17}[\dX]",text)  
print(data_list) # ['130429191912015219']  
"""  
身份则前17位是数字  
[abc]代表中间是a或者b或者是c  
所以这里就是[\dX]代表是数字或者是X  
"""  
  
text = "dsf130429191912015219k1304291959121952Xkk"  
data_list = re.findall("\d{17}(\d|X)",text)  
print(data_list) # ['9']  
"""  
这里括号是分组的意思(\d|X)管道符代表数字或者X  
跟[]的区别是,如果或是1个字符,那就是都可以  
如果是要接\dw|Xr这时候[]不支持,[]支持1个字符  
只提取了最后的数字或者X  
"""  
  
text = "dsf130429191912015219k13042919591219521Xkk"  
data_list = re.findall("(\d{17}(\d|X))",text)  
print(data_list) # [('130429191912015219', '9'), ('13042919591219521X', 'X')]  
"""  
这就是提取身份证和最后一位数字  
"""  
  
text = "dsf130429191912015219k13042919591219521Xkk"  
data_list = re.findall("(\d{6})(\d{4})(\d{2})(\d{2})(\d{3})([0-9]|X)",text)  
print(data_list) # [('130429', '1919', '12', '01', '521', '9'), ('130429', '1959', '12', '19', '521', 'X')]  
"""  
这个按照6位4位2位2位3位提取出来身份证内容  
为了可以提取出生 年 月 日 男女  
"""  
  
  
# 3. 手机号  
text = "我的手机号是15133377892,你得手机号是15210587211"  
data_list = re.findall("1[3|4|5|6|7|8|9]\d{9}",text)  
print(data_list) # ['15133377892', '15210587211']  
"""  
1开头的手机号  
[]可以直接写3456789,代表第二位是3-9 data_list = re.findall("1[3456789]\d{9}",text)  
data_list = re.findall("1[3-9]\d{9}",text)  
上面两种都可以  
后边就是9位数字  
"""  
  
  
# 4. 邮箱地址  
text = "楼主太牛逼了,在线想要192030291834@qq.com和xxxxx@live.com谢谢楼主,手机号也可12390095713034,1239132408510934搞起来啊"  
email_list = re.findall("\w+@\w+\.\w+", text)  
print(email_list) # ['在线想要192030291834@qq.com和xxxxx']  
"""  
字母数字下划线汉字,贪婪匹配 加 @符号 加字母数字下划线 加. 加字母数字下划线  
这里前边有汉字,后边也有汉字,所以邮箱前后的内容都会添加,直到@没法识别了所以停止了  
"""  
  
text = "楼主太牛逼了,在线想要192030291834@qq.com和xxxxx@live.com谢谢楼主,手机号也可12390095713034,1239132408510934搞起来啊"  
email_list = re.findall("[a-zA-Z0-9_-]+@[a-zA-Z0-9_-]+\.[a-zA-Z0-9_-]+",text,re.ASCII)  
print(email_list) # ['192030291834@qq.com', 'xxxxx@live.com']  
"""  
a-z代表所有字符,A-Z代表所有字符大写,0-9的数字,_,- 这些都可以  
+ 代表 1个或者n个 加 @后边同样的  
这里的.因为有特殊的含义,代表除了换行符的所有字符,所以需要在前边加转义符\,让它变成.的意思  
注意最后末尾添加+代表多个字符  
"""  
  
text = "楼主太牛逼了,在线想要192030291834@qq.com和xxxxx@live.com谢谢楼主,手机号也可12390095713034,1239132408510934搞起来啊"  
email_list = re.findall("\w+@\w+\.\w+",text,re.ASCII)  
print(email_list) # ['192030291834@qq.com', 'xxxxx@live.com']  
"""  
在re.findall中添加了re.ASCII参数就可以匹配出来结果了  
因为这个参数是组织你的匹配程序匹配中文  
"""  
  
text = "楼主太牛逼了,在线想要192030-2+9.1834@qq-fd.com+12和xxxxx@live.com谢谢楼主,手机号也可12390095713034,1239132408510934搞起来啊"  
email_list = re.findall("(\w+([-+.]\w+)*@\w+([-.]\w+)*\.\w+([-+.]\w+)*)",text,re.ASCII)  
print(email_list) # [('192030-2+9.1834@qq-fd.com+12', '.1834', '-fd', '+12'), ('xxxxx@live.com', '', '', '')]  
"""  
字符数字下划线多个字符 ()*代表可以出现0次到n次  
[-+.]\w+代表是如果后边存在特殊符号类似于-+.,后一定要加一个元素,邮箱最后不能有这个符号  
@ 后边需要有字符等  
后边代表-.后边也需要有字符  
. 后边加字符等  
后出现-.后边也要有字符  
这里运行的时候,可能会有仅仅使用-.无法匹配出来的情况发生,但是也可能不发生  
"""  
  
  
# 5. 补充代码,实现获取页面上的所有评论(已实现),并提取里面的邮箱  
# pip install requests beautifulsoup4  
res = requests.get(  
    url="https://cee.stanford.edu/people/sarah-billington",  
    headers={  
        "User-Agent": 'Mozilla/5.0 (Macintosh; Intel Max OS X 10_14_0) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/80.0.3987.163 Safari/537.36',  
    },  
)  
  
bs_object = BeautifulSoup(res.text, 'html.parser')  
comment_object_list = bs_object.find_all("p", attrs={"class": "reply-content"})  
for comment_object in comment_object_list:  
    text = comment_object.text  
    print(text)  
    # 提取邮箱  
    email_list = re.findall("\w+@\w+\.\w+", text, re.ASCII)  
    if email_list:  
        print(email_list)
```
